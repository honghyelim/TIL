  • Mutable : 변하는 값 → 리스트, 딕셔너리, 집합

  • 클래스의 상속
class cal: #부모 클래스

	first = 2 #클래스 변수
	second = 3

	def __init__(self, first,second): #객체 변수
		self.first = first
		self.second = second
	def setdata(self, first, second):
		self.first = first
		self.second = second

	def add(self):
		result = self.first + self.second
		return result

#자식 클래스
class morecal(cal):
	def pow(self):
		r = self.first ** self.second
		return r

a = morecal(4,2)
print(a.add())
>> 6
print(a.pow())

  1. 내장함수, 외장함수
  • 내장 함수 : 파이썬에 내장된 함수
#filter 함수
def positive(x):
	return x > 0

a = list(filter(positive,[1,-3,2,0,-5,6]))
print(a)

>> [1,2,6]
#map 함수
a = list(map(lambda x : x*2, [1,2,3,4]))
print(a)

>> [2,4,6,8]
